    2. Locate the `workflow` command block in `packages/cli/src/cli/main.ts` (search for `.command("workflow")` or the existing `.command("exec")` / `.command("run-status")` subcommands). Add a new `watch` subcommand:

       ```typescript
       workflow
         .command("watch")
         .description("Stream workflow run status changes until terminal state")
         .requiredOption("--run-id <id>", "Workflow run ID to watch")
         .option("--poll-interval-ms <ms>", "Poll interval in ms", "2000")
         .option("--timeout-ms <ms>", "Max time to wait in ms", "600000")
         .option("--json", "Output JSON")
         .action(async (opts) => {
           const pollIntervalMs = parseOptionalInteger(opts.pollIntervalMs, "--poll-interval-ms") ?? 2000;
           const timeoutMs = parseOptionalInteger(opts.timeoutMs, "--timeout-ms") ?? 600000;
           if (!opts.runId || typeof opts.runId !== "string") {
             fail("invalid_option_value", "--run-id is required");
           }
           const client = await buildClient(); // match pattern used by neighboring workflow subcommands

           let lastStatus: string | null = null;
           let finalStatus: string | null = null;
           const startedAt = Date.now();
           while (true) {
             let run;
             try {
               run = await client.workflows.getRun(opts.runId);
             } catch (err) {
               fail("workflow_run_fetch_failed", `Failed to fetch run ${opts.runId}: ${(err as Error).message}`);
             }
             const status = extractRunStatus(run);
             if (status && status !== lastStatus) {
               process.stdout.write(JSON.stringify({
                 ts: new Date().toISOString(),
                 run_id: opts.runId,
                 status,
               }) + "\n");
               lastStatus = status;
             }
             if (status && isTerminalRunStatus(status)) {
               finalStatus = status;
               break;
             }
             if (Date.now() - startedAt >= timeoutMs) {
               fail("workflow_watch_timeout", `Run ${opts.runId} did not reach terminal state within ${timeoutMs}ms`);
             }
             await sleep(pollIntervalMs);
           }

           const failed = finalStatus ? isFailedRunStatus(finalStatus) : false;
           printResult({
             schema: WORKFLOW_WATCH_SCHEMA_VERSION,
             run_id: opts.runId,
             final_status: finalStatus,
             success: !failed,
             _links: {
               run: webUrl("workflowRun", { workspaceId: client.sdk.workspaceId, runId: opts.runId }),
             },
           });
           if (failed) process.exitCode = 1;
           // Do NOT call process.exit() — per RESEARCH pitfall #4 (stdout flush race)
         });
       ```

If no run URL key exists, omit `_links.run` and add `_links: {}`.
