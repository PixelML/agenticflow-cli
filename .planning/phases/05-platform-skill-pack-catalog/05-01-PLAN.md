```typescript
// packages/cli/src/cli/platform-catalog.ts — contracts the executor MUST export

export interface PlatformPack {
  name: string;            // pack directory name, e.g. "security-pack"
  description: string;     // from pack.yaml description
  skill_count: number;     // pack.yaml skills.length
  version?: string;
  install_source: string;  // "github:PixelML/skills/packs/<name>" — accepted by parsePackSource()
  _links: {
    browse: string;        // https://github.com/PixelML/skills/tree/main/packs/<name>
  };
}

export interface PlatformSkill {
  name: string;            // skill name from pack.yaml skills[].name
  description: string;     // skill description
  pack: string;            // owning pack name
}

export class PlatformCatalogError extends Error {
  code: 'RATE_LIMITED' | 'NETWORK' | 'PARSE' | 'NOT_FOUND';
  hint: string;            // user-facing hint URL
  constructor(code: PlatformCatalogError['code'], message: string, hint: string);
}

export function fetchPlatformPacks(opts?: { token?: string }): Promise<PlatformPack[]>;
export function fetchPlatformSkills(opts?: { token?: string }): Promise<PlatformSkill[]>;
```
</interfaces>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI → GitHub API | Untrusted JSON + YAML response crosses into the CLI process |
| User shell → CLI | GITHUB_TOKEN env var (secret) read from process.env |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | platform-catalog.ts YAML parser | mitigate | Use existing `yaml` package (safe parser, no JS-YAML !!js/* tags). Wrap `yamlParse` in try/catch and skip malformed packs rather than crashing |
| T-05-02 | Information Disclosure | GITHUB_TOKEN env var | mitigate | Only read via `opts.token` injection — never log it, never include it in PlatformCatalogError messages or hint strings |
| T-05-03 | Denial of Service | GitHub rate limit (60/hr unauth) | mitigate | Catch 403/429 → typed PlatformCatalogError with actionable hint pointing to browser URL + GITHUB_TOKEN suggestion. No retry storm |
| T-05-04 | Spoofing | raw.githubusercontent.com responses | accept | HTTPS-only URLs, GitHub-controlled CA. Pack contents are non-executable metadata (name/description/skills list) — installation is a separate flow |
| T-05-05 | Elevation of Privilege | YAML response causing prototype pollution | mitigate | `yaml` package safe-parses by default. Treat parsed result as `unknown`, narrow each field with explicit `typeof` checks before constructing PlatformPack |
</threat_model>
