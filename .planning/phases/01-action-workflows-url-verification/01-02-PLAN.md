The workflow has two nodes:
1. **draft-response** (node_type_name: "llm") — Generates a professional review reply using Gemma 4
2. **post-to-gbp** (node_type_name: "mcp_run_action") — Posts the reply to Google Business Profile

Full JSON content:
```json
{
  "name": "skill-amazon-post-review-to-gbp",
  "description": "Draft a professional reply to a Google Business Profile review and post it directly. Requires Google Business Profile MCP connection.",
  "nodes": [
    {
      "name": "draft-response",
      "node_type_name": "llm",
      "input_config": {
        "model": "agenticflow/gemma-4-31b-it",
        "system_message": "You are a professional customer experience manager for an Amazon seller who also has a Google Business Profile. Write helpful, empathetic, and concise review responses. Acknowledge specific points from the review. Keep responses under 150 words. Never be defensive or argumentative.",
        "human_message": "Draft a professional response to this Google Business Profile review:\n\nRating: {{review_rating}}/5\nReview text: {{review_text}}\n\nRespond directly — this will be posted as-is."
      }
    },
    {
      "name": "post-to-gbp",
      "node_type_name": "mcp_run_action",
      "input_config": {
        "action": "google_business_profile-reply_to_review",
        "input_params": {
          "account_id": "{{gbp_account_id}}",
          "location_id": "{{gbp_location_id}}",
          "review_id": "{{review_id}}",
          "comment": "${draft-response.generated_text}"
        }
      }
    }
  ],
  "input_schema": {
    "type": "object",
    "title": "GBP Review Reply Input",
    "required": ["review_text", "review_rating", "gbp_account_id", "gbp_location_id", "review_id"],
    "properties": {
      "review_text": { "type": "string", "title": "Review Text", "description": "The full text of the review to respond to" },
      "review_rating": { "type": "integer", "title": "Star Rating (1-5)", "description": "The star rating of the review" },
      "gbp_account_id": { "type": "string", "title": "GBP Account ID", "description": "Google Business Profile account ID" },
      "gbp_location_id": { "type": "string", "title": "GBP Location ID", "description": "Google Business Profile location ID" },
      "review_id": { "type": "string", "title": "Review ID", "description": "The review ID to reply to" }
    }
  },
  "output_mapping": {
    "draft_response": "${draft-response.generated_text}",
    "post_result": "${post-to-gbp.result}",
    "post_success": "${post-to-gbp.success}"
  }
}
```
